/**
 * Simple Example of a program that uses a 'finish' to ensure determinacy.
 */

public class Finish1 {
  var a : Int = 0;
  var b : Int = 1;
  def doit() {
    finish { 
       async { a = 10; }
       async { b = 20; }
    }
    Console.OUT.println("a=" + a + "; b=" + b);
  }
  public static def main(argv:Array[String](1)) {
    (new Finish1()).doit();
  }
}
