/**
 * Simple Example of an 'async' expression.
 */

public class Async1 {
  public static def main(argv:Array[String]) {
     async{ Console.OUT.print("a"); }
     async{ Console.OUT.print("b"); }
     async{ Console.OUT.print("c"); }
  }
}
