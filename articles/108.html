/**
 * Simple Example of an &#39;at&#39; expression.
 */

public class AtExp {
  public static def main(argv:Array[String]) {
    val start = here;
    for(p in Place.places()) {
      at(p) {
        val q = at(p) here;
        assert p == q;
      }
    }
  }
}
